== Example

[source,yaml]
----
download:
  http:
    my_downloader:
      assets:
        - input: https://github.com/jreleaser/jreleaser/raw/main/src/media/icon_256x256.png
          output: '{{projectName}}.png'
        - input: https://github.com/jreleaser/jreleaser/releases/download/v1.0.0/jreleaser-{{projectVersion}}.zip
          unpack:
            enabled: true
----
