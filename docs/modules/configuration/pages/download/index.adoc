= Download

Artifacts and files may be downloaded from configured locations.

include::partial$legend.adoc[]

[tabs]
====
YAML::
+
--
[source,yaml]
[subs="+macros"]
----
# icon:dot-circle[]
download:
  # Enable or disable all configured downloaders.
  # Defaults to `true`.
  # icon:dot-circle[]
  enabled: true
----
--
TOML::
+
--
[source,toml]
[subs="+macros"]
----
# icon:dot-circle[]
[download]

  # Enable or disable all configured downloaders.
  # Defaults to `true`.
  # icon:dot-circle[]
  enable = true
----
--
JSON::
+
--
[source,json]
[subs="+macros"]
----
{
  // icon:dot-circle[]
  "download": {
    // Enable or disable all configured downloaders.
    // Defaults to `true`.
    // icon:dot-circle[]
    "enabled": true
  }
}
----
--
Maven::
+
--
[source,xml]
[subs="verbatim,+macros"]
----
<jreleaser>
  <!--
    icon:dot-circle[]
  -->
  <download>
    <!--
      Enable or disable all configured downloaders.
      Defaults to `true`.
      icon:dot-circle[]
    -->
    <enabled>true</enabled>
  </download>
</jreleaser>
----
--
Gradle::
+
--
[source,groovy]
[subs="+macros"]
----
jreleaser {
  // icon;dot-circle[]
  download {
    // Enable or disable all configured downloaders.
    // Defaults to `true`.
    // icon:dot-circle[]
    enabled = true
  }
}
----
--
